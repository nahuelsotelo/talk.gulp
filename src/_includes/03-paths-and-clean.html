var del = require('del');

gulp.task('clean', function(callback) {
    del([
        basePath.dist,
        basePath.tmp
      ],
      function(err, deletedFiles) {
        console.log('Files deleted:\n'.bold.green , deletedFiles.join(',\n '));
        callback();
      }
    );
});
