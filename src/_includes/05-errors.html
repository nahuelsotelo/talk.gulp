Gulp Plumber


gulp.task('css', function() {
  return gulp.src( path.stylesSrc + '/*.scss' )
    .pipe(plumber({ errorHandler: onError }))
    .pipe(sass())
    .pipe(gulp.dest( path.stylesDist + '/css' ))
})
