Modernizr

Dependecies

  • gulp-modernizr

gulp.task('modernizr', function() {
  gulp.src([
    assetsPath.stylesDist + '/*.css',
    assetsPath.scriptsDist + '/scrpts.min.js',
  ])
  .pipe(modernizr({
    'options': [
      'setClasses',
    ]
  }))
  .pipe(gulp.dest( assetsPath.scriptsDist + '/vendor' ));
});
