Tips para mejorar el workflow

  • Cuidar la velocidad de compilacion. Nos tiene que ahorrar tiempo no quitarnoslo.
  • Cuidado con las dependencias. No depender 100% de un plugin.
  • Abstraction Hell. El código se tiene que seguir entendiendo. Acordar con el equipo.

Gulp 101

Setup

  1. Instalar node
  2. npm install -g gulp
  3. En el proyecto crear:
    • package.json, para las dependencias
    • gulpfile, para configurar los tasks
  4. Instalar y configurar tasks

Sintaxis Básica


var gulp          = require('gulp');
var sass          = require('gulp-sass');
var autoprefixer  = require('gulp-autoprefixer');

gulp.task('css', function() {
  return gulp.src( './scss/*.scss' )
    .pipe(sass())
    .pipe(autoprefixer({
      browsers: [ 'last 2 versions', '> 1%' ]
    }))
    .pipe(gulp.dest( './css' ))
});

Clean


var del = require('del');

gulp.task('clean', function(callback) {
    del([
        basePath.dist,
        basePath.tmp
      ],
      function(err, deletedFiles) {
        console.log('Files deleted:\n'.bold.green , deletedFiles.join(',\n '));
        callback();
      }
    );
});

I'm Watching you

Watch


gulp.task('watch', function() {
    gulp.watch( path.src + '/scss/**/*.scss',       ['css'] );
    gulp.watch( path.src + '/js/**/*.js',           ['js'] );
    gulp.watch( path.src + '/img/**/*.+(png|jpg)',  ['images'] );
    gulp.watch( path.src + '/img/**/*.svg',         ['svg'] );
});

Browsersync

Dependecies

  • browsersync

var browserSync = require('browser-sync').create();

gulp.task('serve', ['build'], function() {
    browserSync({
      server: basePath.dist,
      port: 8000,
      files: [ assetsPath.stylesDist + '/*.css']
    });

    gulp.watch( assetsPath.stylesSrc '/**/*.scss', ['css']);
    gulp.watch( assetsPath.scriptsSrc '/**/*.js', ['js']);
    gulp.watch( basePath.dist + /**/*.html').on('change', browserSync.reload);
    gulp.watch( assetsPath.scriptsDist + /**/*.js').on('change', browserSync.reload);
});
